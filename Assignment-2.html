import numpy as np
import matplotlib.pyplot as plt


# Conditions for outliers
con1 = visa_df['no_of_employees'] < LB
con2 = visa_df['no_of_employees'] > UB
con = con1 | con2

# Calculate the median
median = visa_df['no_of_employees'].median()

# Replace outliers with the median
new_data = np.where(con, median, visa_df['no_of_employees'])

# Ensure visa_df_copy is a copy of visa_df
visa_df_copy = visa_df.copy()
visa_df_copy['no_of_employees'] = new_data

# Plot the boxplots
plt.figure(figsize=(12, 6))

plt.subplot(1, 2, 1)
plt.boxplot(visa_df['no_of_employees'])
plt.title('Original Data')

plt.subplot(1, 2, 2)
plt.boxplot(new_data)
plt.title('Data with Outliers Replaced by Median')

plt.show()
